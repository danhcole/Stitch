//Stitch comments are similar to C comments
//this is a single line comment
/*
You can also have multi-line comments
*/

//functions are declared using the 'def' keyword, like Python
def int main() {

	//the var keyword declares a variable
	var int x = 7; 
	//booleans are a primitive data type
	var bool b = true; 

	//strings are first class citizens in Stitch
	var string s = "This is a String\n";
	var string h = "Stitch also supports " + "string concatenation!";
	var unsigned long l = lengthof(h);

	let double PI = 3.14; //let is used to define constants

	return 0;	
}
